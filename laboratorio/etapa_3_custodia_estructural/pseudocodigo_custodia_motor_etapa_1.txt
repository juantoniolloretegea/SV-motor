  C_custodia^9 = I_CUSTODIA(observables)
  C_gob^9      = Γbar_H(C_custodia^9, H_CUSTODIA)
  A_CUSTODIA   = gate_vector(C_gob^9, C_custodia^9)   # ∈ Σ⁹
  K3           = κ3(C_gob^9, A_CUSTODIA)

  si K3 = APTO:
      política = PERMITIR_AVANCE_CONTROLADO
  si K3 = INDETERMINADO:
      política = DETENER_Y_REVISAR

      política = BLOQUEAR_AVANCE

  devolver trazabilidad, obligaciones y política.
